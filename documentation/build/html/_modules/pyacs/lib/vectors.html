[docs]def vector_product(A,B):
###################################################################
    """
    Vector product
    
                          A[1]*B[2] - B[1]*A[2]
    vector_product(A,B)=  A[2]*B[0] - B[2]*A[0]
                          A[0]*B[1] - B[0]*A[1]
                          
    :param A,B: 1D numpy arrays
    
    :return AxB: 1D numpy array
    """
    
    import numpy as np
    
    C=np.zeros([3])
    C[0]=A[1]*B[2]-B[1]*A[2]
    C[1]=A[2]*B[0]-B[2]*A[0]
    C[2]=A[0]*B[1]-B[0]*A[1]
    
    return(C)
